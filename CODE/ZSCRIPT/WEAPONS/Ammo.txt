Class z_TNTAmmoBox : CustomInventory
{

	default
	{
		inventory.pickupmessage "$TNTBOX_PICKED";
		inventory.pickupsound "dsammoup";
		//inventory.amount 5;
		//inventory.maxamount 50;
		inventory.icon "TBNDA0";
		TAG "$TNTBOX_NAME";
		-NOGRAVITY;
		scale 0.75;
		+SHOOTABLE;
		+NOBLOOD;
		+NOBLOODDECALS;
		+INVULNERABLE;

	}
	Override void PostBeginPlay()
	{
		Super.PostBeginPlay();
		GiveInventory("Pointable", 1);
	}
	States
	{
		Spawn:
			TBOX A -1;
			Stop;
		Pickup:
			TNT1 A 0 A_GiveInventory("TNTAmmo", 5);
			TNT1 A 0 A_GiveInventory("Dynamite", 1);
			Stop;
	}
}
