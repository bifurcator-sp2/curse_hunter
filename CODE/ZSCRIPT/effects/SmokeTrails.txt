Class z_raiserTrial : Actor
{
	int spin;
	default
	{
	  Scale 0.25;
	  Alpha 0.5;
	  +NOINTERACTION;
	  +NOGRAVITY;
	  RENDERSTYLE "TRANSLUCENT";
	  +ClientSideOnly;
	  +FORCEXYBILLBOARD;
	  +ROLLSPRITE;
	 }
	Override void PostBeginPlay(){
		Super.PostBeginPlay();
		spin = frandom(-8, 8);
	}
	 States
	 {
		Spawn:
			S2MK A 1 {
				A_fadeOut(0.01, FTF_REMOVE);
				A_SetRoll(roll+spin,SPF_INTERPOLATE);
			}
			Loop;

	 }

}


class z_BurnDeathSmoke : actor
{
	int counter;

	default
	{
		+NOINTERACTION;
		+DONTSPLASH;
		+NOCLIP;
		+NOGRAVITY;
		+NOBLOCKMAP;
		+CLIENTSIDEONLY;
		RenderStyle "Translucent";
		Alpha 0.5;
		scale 0.5;
	}

	override void Tick(){
		super.tick();
		if(Master){
			vector3 MyPos =  Master.pos;
			MyPos.z = Master.z+Master.height;
			setOrigin(MyPos, 1);
			if(counter<35*15){
				counter++;
			} else {
				a_fadeOut(0.005);
			}
		} else {
			destroy();
		}
	}

	States
	{
		Spawn:
			BDSM ABCD 3;
			Loop;
	}

}
